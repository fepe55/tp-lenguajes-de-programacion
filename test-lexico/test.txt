procedure DIVIDIR;
var W;
begin
R := X;
Q := 0;

while W <= R do W := W * 2;
while W > Y do
begin
Q := Q * 2;
W := W / 2;
if W <= R then
begin
R := R - W;
Q := Q + 1
end
end
end;
